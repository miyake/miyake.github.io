\begin{align} \frac{\partial f}{\partial x} &= \frac{1}{x+y^2} \\ \frac{\partial f}{\partial y} &= \frac{2y}{x+y^2} \\ \frac{\partial^2 f}{\partial x^2} &= - \frac{1}{\left( x+y^2 \right)^2} \\ \frac{\partial^2 f}{\partial x \partial y} &= - \frac{2y}{\left( x+y^2 \right)^2} \\ \frac{\partial^2 f}{\partial y^2} &= \frac{2 \left( x - y^2 \right)}{\left( x+y^2 \right)^2} \end{align}
