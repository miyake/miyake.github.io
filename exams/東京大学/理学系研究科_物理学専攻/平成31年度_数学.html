\begin{align} \left( i [A, B] \right)^\dagger &= \left( i \left( AB - BA \right) \right)^\dagger \\ &= -i \left( B^\dagger A^\dagger - A^\dagger B^\dagger \right) \\ &= -i \left( BA - AB \right) \\ &= i \left( AB - BA \right) \\ &= i [A, B] \\ \\ \text{tr} \left( i [A, B] \right) &= i \left( \text{tr} (AB) - \text{tr} (BA) \right) \\ &= i \left( \text{tr} (AB) - \text{tr} (AB) \right) \\ &= 0 \end{align}
